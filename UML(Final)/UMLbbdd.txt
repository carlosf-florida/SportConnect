@startuml
hide circle
skinparam classAttributeIconSize 0

' =======================
' ENTIDADES
' =======================

class usuarios {
  +id : int
  nombre : varchar
  apellido : varchar
  nick : varchar
  email : varchar
  password : varchar
  rol_id : int
  fecha_creacion : timestamp
}

class roles {
  +id : int
  nombre : varchar
}

class clases {
  +id : int
  titulo : varchar
  descripcion : text
  deporte : varchar
  profesor_id : int
  es_premium : boolean
  precio : decimal
  fecha_creacion : timestamp
}

class asistencias {
  +id : int
  usuario_id : int
  clase_id : int
  fecha : datetime
  asistio : boolean
}

class mensajes {
  +id : int
  remitente_id : int
  clase_id : int
  contenido : text
  fecha_envio : timestamp
}

class pagos {
  +id : int
  usuario_id : int
  clase_id : int
  monto : decimal
  fecha_pago : timestamp
}

class productos {
  +id : int
  nombre : varchar
  descripcion : text
  precio : decimal
  stock : int
  vendedor_id : int
  fecha_publicacion : timestamp
}

class pedidos {
  +id : int
  comprador_id : int
  producto_id : int
  cantidad : int
  total : decimal
  fecha_pedido : timestamp
}

class suscripciones {
  +id : int
  usuario_id : int
  clase_id : int
  fecha_suscripcion : timestamp
}

class valoraciones {
  +id : int
  usuario_id : int
  clase_id : int
  puntuacion : int
  comentario : text
  fecha : timestamp
}

' =======================
' RELACIONES
' =======================

roles "1" <-- "0..*" usuarios : rol_id

usuarios "1" <-- "0..*" clases : profesor_id

usuarios "1" <-- "0..*" asistencias : usuario_id
clases "1" <-- "0..*" asistencias : clase_id

usuarios "1" <-- "0..*" mensajes : remitente_id
clases "1" <-- "0..*" mensajes : clase_id

usuarios "1" <-- "0..*" pagos : usuario_id
clases "1" <-- "0..*" pagos : clase_id

usuarios "1" <-- "0..*" productos : vendedor_id

usuarios "1" <-- "0..*" pedidos : comprador_id
productos "1" <-- "0..*" pedidos : producto_id

usuarios "1" <-- "0..*" suscripciones : usuario_id
clases "1" <-- "0..*" suscripciones : clase_id

usuarios "1" <-- "0..*" valoraciones : usuario_id
clases "1" <-- "0..*" valoraciones : clase_id
